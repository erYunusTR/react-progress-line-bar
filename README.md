## Install

```bash
npm install --save react-progress-line-bar
```

## Usage

```jsx
import React from 'react'
import ReactProgressLineBar from 'react-progress-line-bar'

const App = () => {
  return (
    <div>
      <ReactProgressLineBar value={60}/>
    </div>
  )
}

export default App
```
